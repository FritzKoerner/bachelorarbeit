    """
    CoordinateLandingEnv: Genesis-batched drone environment for coordinate-targeted landing.

    Returns torch tensors natively. SB3 wrapper (Phase 3) converts to numpy.
    """
    import torch
    import genesis as gs
    from controllers.velocity_controller import VelocityController


    class CoordinateLandingEnv:
        def __init__(self, config: dict, device: torch.device = None):
            # Config extraction ONLY — no Genesis calls here
            self.n_envs = config['env']['n_envs']
            self.max_steps = config['env']['max_episode_steps']
            self.show_viewer = config['scene'].get('show_viewer', False)
            self.env_spacing = config['scene'].get('env_spacing', 80.0)

            # Spawn parameters
            self.height_min = config['spawn']['height_min']    # 10.0
            self.height_max = config['spawn']['height_max']    # 20.0
            self.target_range = config['spawn']['target_range']  # 30.0
            self.drone_offset_range = config['spawn'].get('drone_offset_range', 10.0)

            # Action scaling
            self.vel_scale = config['controller'].get('vel_scale', 2.0)
            self.yaw_scale = config['controller'].get('yaw_scale', 1.0)

            self.device = device or torch.device('cuda' if torch.cuda.is_available() else 'cpu')
            self._built = False

        def build(self):
            # Genesis scene creation — called AFTER gs.init() in smoke_test.py
            scene = gs.Scene(
                sim_options=gs.options.SimOptions(dt=0.01, substeps=2),
                viewer_options=gs.options.ViewerOptions(
                    camera_pos=(0, -50, 40),
                    camera_lookat=(0, 0, 0),
                    camera_fov=60,
                ),
                show_viewer=self.show_viewer,
            )

            # Flat ground — single plane, no terrain features
            scene.add_entity(gs.morphs.Plane())

            # Drone from URDF — path relative to project root (cwd)
            # assets/robots is a symlink to prototyp_2/assets/robots
            self.drone = scene.add_entity(
                gs.morphs.Drone(
                    file="assets/robots/draugas/draugas_genesis.urdf",
                    pos=(0, 0, 15.0),
                    euler=(0, 0, 90),
                    propellers_link_name=["prop0_link", "prop1_link", "prop2_link", "prop3_link"],
                    propellers_spin=[1, -1, 1, -1],
                )
            )

            scene.build(n_envs=self.n_envs, env_spacing=(self.env_spacing, self.env_spacing))
            self.scene = scene

            # Target positions: plain torch tensors, no physics entity needed
            self.target_pos = torch.zeros(self.n_envs, 3, device=self.device)

            # Episode state
            self.step_count = torch.zeros(self.n_envs, dtype=torch.long, device=self.device)

            # Velocity controller (per research: port verbatim from prototyp_2)
            controller_cfg = {}  # populated if config passed through; VelocityController has sensible defaults
            self.controller = VelocityController(device=self.device)

            self._built = True

        def reset(self, env_ids: torch.Tensor = None) -> torch.Tensor:
            """
            Reset environments.

            Args:
                env_ids: (k,) int tensor of env indices to reset, or None to reset all.

            Returns:
                obs: (n_envs, 13) or (k, 13) float32 tensor
            """
            if env_ids is None:
                env_ids = torch.arange(self.n_envs, device=self.device)

            n_reset = len(env_ids)

            # Sample random target positions: x,y in [-target_range, +target_range], z=0
            target_x = (torch.rand(n_reset, device=self.device) - 0.5) * 2 * self.target_range
            target_y = (torch.rand(n_reset, device=self.device) - 0.5) * 2 * self.target_range
            self.target_pos[env_ids] = torch.stack(
                [target_x, target_y, torch.zeros(n_reset, device=self.device)], dim=-1
            )

            # Sample drone spawn: random x/y offset from target, 10-20m altitude
            offset_x = (torch.rand(n_reset, device=self.device) - 0.5) * 2 * self.drone_offset_range
            offset_y = (torch.rand(n_reset, device=self.device) - 0.5) * 2 * self.drone_offset_range
            spawn_z = torch.rand(n_reset, device=self.device) * (self.height_max - self.height_min) + self.height_min

            spawn_pos = torch.stack([target_x + offset_x, target_y + offset_y, spawn_z], dim=-1)

            # Random yaw for spawn orientation
            yaw = torch.rand(n_reset, device=self.device) * 2 * torch.pi
            spawn_quat = self._yaw_to_quat(yaw)

            # set_pos with zero_velocity=True (default) — zeroes all dof velocities
            # Do NOT call set_vel or set_ang_vel — these methods do not exist in Genesis 0.3.13
            self.drone.set_pos(spawn_pos, envs_idx=env_ids)
            self.drone.set_quat(spawn_quat, envs_idx=env_ids)

            self.step_count[env_ids] = 0
            self.controller.reset()

            # One physics step to settle state
            self.scene.step()

            return self._get_obs()

        def step(self, actions: torch.Tensor):
            """
            Step the environment.

            Args:
                actions: (n_envs, 4) float in [-1, 1]: [vx, vy, vz, yaw_rate]

            Returns:
                obs: (n_envs, 13) torch.Tensor
                rewards: (n_envs,) torch.Tensor  -- zeros in Phase 1; Phase 2 adds reward shaping
                dones: (n_envs,) bool torch.Tensor
                info: dict
            """
            # Scale actions to physical ranges
            target_vel = actions[:, :3] * self.vel_scale       # [-vel_scale, +vel_scale] m/s
            target_yaw_rate = actions[:, 3] * self.yaw_scale   # [-yaw_scale, +yaw_scale] rad/s

            # Current state for controller
            current_vel = self.drone.get_vel()
            current_quat = self.drone.get_quat()

            # Compute and apply RPMs
            rpms = self.controller.compute_rpm(target_vel, target_yaw_rate, current_vel, current_quat)
            self.drone.set_propellels_rpm(rpms)  # note: typo in Genesis API — double l is correct
            self.scene.step()

            self.step_count += 1
            obs = self._get_obs()

            # Termination conditions (rewards populated in Phase 2)
            pos = self.drone.get_pos()
            timeout = self.step_count >= self.max_steps
            crashed = pos[:, 2] < 0.1  # drone below 10cm — hit ground
            dones = timeout | crashed

            rewards = torch.zeros(self.n_envs, device=self.device)

            # Auto-reset done environments
            done_ids = torch.where(dones)[0]
            if len(done_ids) > 0:
                self.reset(done_ids)

            return obs, rewards, dones, {}

        def _get_obs(self) -> torch.Tensor:
            """
            Build 13D observation vector.

            Returns:
                (n_envs, 13) float32: [rel_pos(3), vel(3), quat(4), ang_vel(3)]
            """
            pos = self.drone.get_pos()        # (n_envs, 3)
            vel = self.drone.get_vel()        # (n_envs, 3)
            quat = self.drone.get_quat()      # (n_envs, 4) — [w, x, y, z]
            ang_vel = self.drone.get_ang()    # (n_envs, 3) — CORRECT: get_ang(), NOT get_ang_vel()

            # Relative position: how far is drone from target (Phase 1 key design)
            rel_pos = self.target_pos - pos   # (n_envs, 3)

            state = torch.cat([rel_pos, vel, quat, ang_vel], dim=-1)  # (n_envs, 13)
            return state

        def _yaw_to_quat(self, yaw: torch.Tensor) -> torch.Tensor:
            """
            Convert yaw angle (radians) to quaternion [w, x, y, z].

            Args:
                yaw: (n,) float tensor

            Returns:
                quat: (n, 4) float tensor [w, x, y, z]
            """
            half_yaw = yaw * 0.5
            w = torch.cos(half_yaw)
            x = torch.zeros_like(yaw)
            y = torch.zeros_like(yaw)
            z = torch.sin(half_yaw)
            return torch.stack([w, x, y, z], dim=-1)
    ```

    """
    Phase 1 smoke test: verifies CoordinateLandingEnv can reset, step, and return observations.

    Run from project root:
        python smoke_test.py

    Expected output:
        Smoke test PASSED
          obs shape:  torch.Size([16, 13])
          obs dtype:  torch.float32
          rewards shape: torch.Size([16])
          dones shape:   torch.Size([16])
          dones (first step): 0 of 16 done
    """
    import torch
    import genesis as gs
    import yaml
    from envs.coordinate_landing_env import CoordinateLandingEnv


    def main():
        # Load config
        with open("config/training_config.yaml") as f:
            config = yaml.safe_load(f)

        n_envs = config['env']['n_envs']
        device = torch.device("cuda" if torch.cuda.is_available() else "cpu")

        # gs.init() called ONCE here — never inside CoordinateLandingEnv
        gs.init(backend=gs.gpu if torch.cuda.is_available() else gs.cpu)

        # Build environment
        env = CoordinateLandingEnv(config, device=device)
        env.build()

        # Test reset()
        obs = env.reset()
        assert obs.shape == (n_envs, 13), f"reset() obs shape mismatch: got {obs.shape}, expected ({n_envs}, 13)"
        assert obs.dtype == torch.float32, f"reset() obs dtype mismatch: got {obs.dtype}, expected float32"

        # Test step() with zero actions
        action = torch.zeros(n_envs, 4, device=device)
        obs, rewards, dones, info = env.step(action)
        assert obs.shape == (n_envs, 13), f"step() obs shape mismatch: got {obs.shape}"
        assert rewards.shape == (n_envs,), f"step() rewards shape mismatch: got {rewards.shape}"
        assert dones.shape == (n_envs,), f"step() dones shape mismatch: got {dones.shape}"
        assert obs.dtype == torch.float32, f"step() obs dtype: {obs.dtype}"

        # Test that target_pos is being set (relative pos should not be zero after reset)
        rel_pos = obs[:, :3]
        # At least some environments should have non-trivial relative positions
        assert rel_pos.abs().sum() > 0, "All relative positions are zero — target_pos not being set"

        # Test partial reset (reset subset of envs)
        env_ids = torch.tensor([0, 3, 7], device=device)
        obs_partial = env.reset(env_ids)
        assert obs_partial.shape == (n_envs, 13), \
            f"Partial reset obs shape mismatch: got {obs_partial.shape}"

        print("Smoke test PASSED")
        print(f"  obs shape:     {obs.shape}")
        print(f"  obs dtype:     {obs.dtype}")
        print(f"  rewards shape: {rewards.shape}")
        print(f"  dones shape:   {dones.shape}")
        print(f"  dones (first step): {dones.sum().item()} of {n_envs} done")
        print(f"  target_pos sample: {env.target_pos[0].tolist()}")
        print(f"  rel_pos sample:    {rel_pos[0].tolist()}")


    if __name__ == "__main__":
        main()
    ```
